zabbix监控mysql

# 1.查找并拷贝Zabbix监控Mysql模板文件

[root@new1 mysql]# find / -name user*_mysql.conf
/usr/share/doc/zabbix-agent-5.0.3/userparameter_mysql.conf
[root@new1 mysql]# cp /usr/share/doc/zabbix-agent-5.0.3/userparameter_mysql.conf /etc/zabbix/zabbix_agentd.d/
[root@new1 mysql]# chown -R zabbix:zabbix /etc/zabbix/zabbix_agentd.d/userparameter_mysql.conf
[root@new1 mysql]# chmod a+x -R /etc/zabbix/zabbix_agentd.d/userparameter_mysql.conf
[root@new1 mysql]# vim /var/lib/zabbix/.my.cnf

# 2.配置.my.cnf

[root@new1 zabbix]# cat /var/lib/zabbix/.my.cnf
[mysql]
host=localhost
user=root       
password=hadoop     
socket=/home/smartmining/mysql/mysql.sock

[mysqladmin]
host=localhost
user=root
password=hadoop
socket=/home/smartmining/mysql/mysql.sock

# 3.重启agent
